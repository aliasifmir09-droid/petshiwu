### Product Schema
```typescript
{
  name: string,
  description: string,
  brand: string,
  category: ObjectId,
  images: string[],
  variants: [{
    size/weight: string,
    price: number,
    stock: number,
    sku: string
  }],
  basePrice: number,
  petType: string,
  tags: string[],
```
